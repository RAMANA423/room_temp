    <div id="result"></div>

    <script>
        function calculateTemp() {
            let roomTemp = document.getElementById("roomTemp").value;

            if (roomTemp === "") {
                document.getElementById("result").innerHTML = "Please enter a value!";
                return;
            }

            roomTemp = parseFloat(roomTemp);

            // Simple estimation formula
            let pcTemp = roomTemp + 20;

            let status = "";

            if (pcTemp < 50) {
                status = "Cool ✅";
            } else if (pcTemp < 70) {
                status = "Normal ⚠️";
            } else {
                status = "Hot 🔥";
            }

            document.getElementById("result").innerHTML =
                "Estimated PC Temperature: " + pcTemp + "°C<br>Status: " + status;
        }
    </script>
